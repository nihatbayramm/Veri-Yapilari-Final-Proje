CLASS ACNode:
    children <- empty map
    fail <- NULL
    output <- empty set


CLASS AhoCorasick:

    INIT:
        root <- new ACNode
        pattern_count <- 0


    FUNCTION add_pattern(pattern):
        node <- root

        FOR EACH character c IN lowercase(pattern):
            IF c NOT IN node.children:
                node.children[c] <- new ACNode
            node <- node.children[c]

        ADD pattern TO node.output
        pattern_count <- pattern_count + 1


    FUNCTION build():
        queue <- empty queue

        FOR EACH child IN root.children:
            child.fail <- root
            ENQUEUE child

        WHILE queue NOT EMPTY:
            u <- DEQUEUE queue

            FOR EACH (char c, node v) IN u.children:
                f <- u.fail

                WHILE f IS NOT NULL AND c NOT IN f.children:
                    f <- f.fail

                IF f IS NOT NULL AND c IN f.children:
                    v.fail <- f.children[c]
                ELSE:
                    v.fail <- root

                IF v.fail == v:
                    v.fail <- root

                v.output <- v.output UNION v.fail.output

                ENQUEUE v


    FUNCTION search(text):
        node <- root
        results <- empty list
        text <- lowercase(text)

        FOR i FROM 0 TO length(text)-1:
            c <- text[i]

            WHILE node != root AND c NOT IN node.children:
                node <- node.fail

            IF c IN node.children:
                node <- node.children[c]

            FOR EACH pattern IN node.output:
                start <- i - length(pattern) + 1
                ADD (start, pattern) TO results

        RETURN results


    FUNCTION inspect_packet(payload):
        RETURN search(payload)


    FUNCTION stats():
        RETURN pattern_count



FUNCTION KMP_MULTI(text, patterns):

    FUNCTION failure(p):
        f <- array of zeros (size len(p))
        k <- 0

        FOR i FROM 1 TO len(p)-1:
            WHILE k > 0 AND p[k] != p[i]:
                k <- f[k-1]

            IF p[k] == p[i]:
                k <- k + 1

            f[i] <- k

        RETURN f


    results <- empty list
    text <- lowercase(text)

    FOR EACH pattern IN patterns:
        p <- lowercase(pattern)
        f <- failure(p)
        q <- 0

        FOR i FROM 0 TO length(text)-1:
            WHILE q > 0 AND p[q] != text[i]:
                q <- f[q-1]

            IF p[q] == text[i]:
                q <- q + 1

            IF q == length(p):
                ADD (i - length(p) + 1, pattern) TO results
                q <- f[q-1]

    RETURN results
