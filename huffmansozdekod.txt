CLASS HuffNode:
    byte_val
    freq
    left
    right

    FUNCTION __lt__(other):
        RETURN self.freq < other.freq


FUNCTION build_huffman_tree(data):
    freq_table <- frequency count of bytes in data
    heap <- empty priority queue

    FOR EACH (byte, freq) IN freq_table:
        ADD new HuffNode(byte, freq) TO heap

    heapify(heap)

    WHILE size(heap) > 1:
        left <- pop smallest freq node from heap
        right <- pop next smallest freq node from heap

        parent <- new HuffNode(NULL, left.freq + right.freq)
        parent.left <- left
        parent.right <- right

        push parent INTO heap
    END WHILE

    IF heap IS NOT EMPTY:
        RETURN heap[0]
    ELSE:
        RETURN NULL


FUNCTION generate_codes(node, prefix, table):
    IF table IS NULL:
        table <- empty map

    IF node IS NULL:
        RETURN table

    IF node.byte_val IS NOT NULL:
        IF prefix IS EMPTY:
            table[node.byte_val] <- "0"
        ELSE:
            table[node.byte_val] <- prefix

    ELSE:
        generate_codes(node.left,  prefix + "0", table)
        generate_codes(node.right, prefix + "1", table)

    RETURN table


FUNCTION encode(data, table):
    result <- empty string

    FOR EACH byte IN data:
        result <- result + table[byte]

    RETURN result


FUNCTION decode(bit_string, root):
    result <- empty list
    node <- root

    FOR EACH bit IN bit_string:
        IF bit == '0':
            node <- node.left
        ELSE:
            node <- node.right

        IF node.byte_val IS NOT NULL:
            APPEND node.byte_val TO result
            node <- root

    RETURN bytes(result)


FUNCTION vpn_compress(payload):
    start_time <- now()

    tree <- build_huffman_tree(payload)
    table <- generate_codes(tree, "", empty map)
    encoded <- encode(payload, table)

    elapsed_ms <- now() - start_time

    original_bits <- length(payload) * 8
    compressed_bits <- length(encoded)

    ratio <- (1 - compressed_bits / original_bits) * 100

    RETURN tree, table, encoded, elapsed_ms,
           original_bits, compressed_bits, ratio


FUNCTION vpn_decompress(encoded, tree):
    start_time <- now()

    decoded <- decode(encoded, tree)

    elapsed_ms <- now() - start_time

    RETURN decoded, elapsed_ms
