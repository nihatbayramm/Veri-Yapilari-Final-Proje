FUNCTION compute_failure(pattern):
    f <- array of size length(pattern) filled with 0
    k <- 0

    FOR i <- 1 TO length(pattern)-1:
        WHILE k > 0 AND pattern[k] != pattern[i]:
            k <- f[k - 1]

        IF pattern[k] == pattern[i]:
            k <- k + 1

        f[i] <- k
    END FOR

    RETURN f


FUNCTION kmp_search(text, pattern):
    IF pattern IS EMPTY:
        RETURN empty list

    f <- compute_failure(pattern)
    matches <- empty list
    q <- 0

    FOR i, c IN text:
        WHILE q > 0 AND pattern[q] != c:
            q <- f[q - 1]

        IF pattern[q] == c:
            q <- q + 1

        IF q == length(pattern):
            ADD (i - length(pattern) + 1) TO matches
            q <- f[q - 1]

    RETURN matches


FUNCTION naive_search(text, pattern):
    matches <- empty list
    n <- length(text)
    m <- length(pattern)

    FOR i <- 0 TO n - m:
        IF text[i : i + m] == pattern:
            ADD i TO matches

    RETURN matches


FUNCTION scan_log_file(log_lines, signatures):
    alerts <- empty list

    FOR lineno <- 1 TO length(log_lines):
        line <- log_lines[lineno]
        normalized <- lowercase(line)

        FOR EACH sig IN signatures:
            hits <- kmp_search(normalized, lowercase(sig))

            IF hits IS NOT EMPTY:
                ADD {
                    line: lineno,
                    signature: sig,
                    positions: hits,
                    content: first 80 chars of line
                } TO alerts

    RETURN alerts


FUNCTION generate_fake_log(num_lines, inject_attacks):
    normal <- list of safe HTTP log strings
    attacks <- list of malicious payload strings

    lines <- empty list

    FOR i <- 1 TO num_lines:
        ADD random choice from normal TO lines

    IF inject_attacks = TRUE:
        positions <- random sample of line indexes

        FOR EACH pos IN positions:
            lines[pos] <- random attack string

    RETURN lines
