FUNCTION reverse_domain(domain):
    domain <- lowercase(domain)
    parts <- split(domain, ".")
    reverse(parts)
    RETURN join(parts, ".")


FUNCTION insert(node, word, idx, threat_type):
    c <- word[idx]

    IF node IS NULL:
        node <- NEW TSTNode(c)
        node_count <- node_count + 1

    IF c < node.c:
        node.left <- insert(node.left, word, idx, threat_type)

    ELSE IF c > node.c:
        node.right <- insert(node.right, word, idx, threat_type)

    ELSE:
        IF idx + 1 = length(word):
            node.is_malicious <- TRUE
            node.threat_type <- threat_type
        ELSE:
            node.mid <- insert(node.mid, word, idx + 1, threat_type)

    RETURN node


FUNCTION add_threat(domain, threat_type):
    rev <- reverse_domain(domain)

    root <- insert(root, rev, 0, threat_type)

    domain_count <- domain_count + 1


FUNCTION search(node, word, idx):
    IF node IS NULL OR idx >= length(word):
        RETURN NULL

    c <- word[idx]

    IF c < node.c:
        RETURN search(node.left, word, idx)

    ELSE IF c > node.c:
        RETURN search(node.right, word, idx)

    ELSE:
        IF node.is_malicious = TRUE:
            RETURN node.threat_type

        IF idx + 1 = length(word):
            IF node.is_malicious = TRUE:
                RETURN node.threat_type
            ELSE:
                RETURN NULL

        RETURN search(node.mid, word, idx + 1)


FUNCTION check_domain(domain):
    rev <- reverse_domain(domain)

    parts <- split(rev, ".")

    FOR i <- 0 TO length(parts)-1:
        prefix <- join(parts[0...i], ".")

        result <- search(root, prefix, 0)

        IF result IS NOT NULL:
            RETURN TRUE, result
    END FOR

    RETURN FALSE, NULL


FUNCTION stats():
    RETURN node_count, domain_count
